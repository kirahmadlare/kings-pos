# GitHub Push Commands
# Replace YOUR_USERNAME with your actual GitHub username

# After creating repository on GitHub, run these commands:

cd C:\Users\kingm\Downloads\POS

# Add GitHub remote
git remote add origin https://github.com/YOUR_USERNAME/kings-pos.git

# Verify remote
git remote -v

# Push to GitHub
git branch -M main
git push -u origin main

# Enter your GitHub credentials when prompted
# (You may need to use a Personal Access Token instead of password)

---

# If you get authentication error, create a Personal Access Token:
# 1. Go to https://github.com/settings/tokens
# 2. Click "Generate new token (classic)"
# 3. Select scopes: repo (all), workflow
# 4. Copy the token
# 5. Use the token as your password when pushing
